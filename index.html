function doGet() { return ContentService.createTextOutput("Web App Aktif"); } function doPost(e) { try { // ===== KONFIGURASI ===== const SPREADSHEET_ID = "ISI_ID_SPREADSHEET_ANDA"; const SHEET_NAME = "Sheet1"; const JAM_MULAI = "06:30"; const JAM_SELESAI = "07:15"; const ZONA = "Asia/Jakarta"; // ===== CEK JAM SERVER (WIB) ===== const now = new Date(); const jamWIB = Utilities.formatDate(now, ZONA, "HH:mm"); if (jamWIB < JAM_MULAI || jamWIB > JAM_SELESAI) { return ContentService .createTextOutput("DITOLAK: Di luar jam absen") .setMimeType(ContentService.MimeType.TEXT); } // ===== SIMPAN DATA ===== const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); const nama = e.parameters.nama ? e.parameters.nama[0] : ""; const status = e.parameters.status ? e.parameters.status[0] : ""; const waktu = Utilities.formatDate(now, ZONA, "yyyy-MM-dd HH:mm:ss"); sheet.appendRow([nama, status, waktu]); return ContentService .createTextOutput("OK") .setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService .createTextOutput("ERROR: " + err.message) .setMimeType(ContentService.MimeType.TEXT); } }
